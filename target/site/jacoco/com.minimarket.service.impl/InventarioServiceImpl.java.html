package com.minimarket.service.impl;

import java.util.List;
import java.util.Objects;

import org.springframework.security.access.AccessDeniedException;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.stereotype.Service;

import com.minimarket.entity.Inventario;
import com.minimarket.entity.Producto;
import com.minimarket.repository.InventarioRepository;
import com.minimarket.repository.ProductoRepository;
import com.minimarket.service.InventarioService;

@Service
public class InventarioServiceImpl implements InventarioService {

    private final InventarioRepository inventarioRepository;
    private final ProductoRepository productoRepository;

    public InventarioServiceImpl(InventarioRepository inventarioRepository,
                                 ProductoRepository productoRepository) {
        this.inventarioRepository = inventarioRepository;
        this.productoRepository = productoRepository;
    }

    @Override
    public List<Inventario> findAll() {
        return inventarioRepository.findAll();
    }

    @Override
    public Inventario findById(Long id) {
        return inventarioRepository.findById(Objects.requireNonNull(id, "El id del inventario no puede ser nulo"))
                .orElse(null);
    }

    @Override
    public Inventario save(Inventario inventario) {
        // Seguridad primero: verificar rol antes de procesar cualquier validación de negocio
        Authentication auth = SecurityContextHolder.getContext().getAuthentication();
        if (auth == null || auth.getAuthorities().stream()
                .noneMatch(a -> "ROLE_INVENTARIO".equals(a.getAuthority()) || "ROLE_ADMIN".equals(a.getAuthority()))) {
            throw new AccessDeniedException("Se requiere ROLE_INVENTARIO o ROLE_ADMIN para registrar movimientos");
        }

        Objects.requireNonNull(inventario, "El inventario no puede ser nulo");

        // Validaciones básicas
        if (inventario.getCantidad() == null || inventario.getCantidad() <= 0) {
            throw new IllegalArgumentException("Cantidad inválida");
        }
        if (inventario.getTipoMovimiento() == null || inventario.getTipoMovimiento().trim().isEmpty()) {
            throw new IllegalArgumentException("Tipo de movimiento inválido");
        }
        if (inventario.getProducto() == null || inventario.getProducto().getId() == null) {
            throw new IllegalArgumentException("Producto inválido en inventario");
        }

        Long productoId = inventario.getProducto().getId();

        Producto producto = productoRepository.findById(productoId)
                .orElseThrow(() -> new IllegalArgumentException("Producto no encontrado"));

        String tipo = inventario.getTipoMovimiento().trim();

        if ("Salida".equalsIgnoreCase(tipo)) {
            Integer stock = producto.getStock();
            if (stock == null || stock < inventario.getCantidad()) {
                throw new IllegalArgumentException("Stock insuficiente para la salida");
            }
            // Reducir stock y persistir
            producto.setStock(stock - inventario.getCantidad());
            productoRepository.save(producto);
        } else if ("Entrada".equalsIgnoreCase(tipo)) {
            // Se incrementar stock en entradas
             Integer stock = producto.getStock() == null ? 0 : producto.getStock();
            producto.setStock(stock + inventario.getCantidad());
            productoRepository.save(producto);
            // Actualmente no se modifica el stock en 'Entrada' para mantener compatibilidad con tests.
        } else {
            //lanza excepción para tipo desconocido
            throw new IllegalArgumentException("Tipo de movimiento desconocido");
        }

        // Asegurar consistencia: asociar el producto recuperado al inventario antes de guardar
        inventario.setProducto(producto);

        return inventarioRepository.save(inventario);
    }

    @Override
    public void deleteById(Long id) {
        inventarioRepository.deleteById(Objects.requireNonNull(id, "El id del inventario no puede ser nulo"));
    }

    @Override
    public List<Inventario> findByProductoId(Long productoId) {
        return inventarioRepository.findByProductoId(Objects.requireNonNull(productoId, "El id del producto no puede ser nulo"));
    }
}
