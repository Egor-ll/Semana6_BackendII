package com.minimarket.service.impl;

import com.minimarket.entity.Usuario;
import com.minimarket.repository.UsuarioRepository;
import com.minimarket.service.AuthService;
import com.minimarket.service.ActividadSeguridadService;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;

@Service
public class AuthServiceImpl implements AuthService {

    private final UsuarioRepository usuarioRepository;
    private final PasswordEncoder passwordEncoder;
    private final ActividadSeguridadService auditoria; // Tu clase de auditoría

    public AuthServiceImpl(UsuarioRepository usuarioRepository, 
                           PasswordEncoder passwordEncoder,
                           ActividadSeguridadService auditoria) {
        this.usuarioRepository = usuarioRepository;
        this.passwordEncoder = passwordEncoder;
        this.auditoria = auditoria;
    }

    @Override
    public Usuario authenticate(String username, String password) {
        try {
            Usuario usuario = usuarioRepository.findByUsername(username)
                    .orElseThrow(() -> new IllegalArgumentException("Usuario no encontrado"));

            if (!passwordEncoder.matches(password, usuario.getPassword())) {
                auditoria.registrar(username, "LOGIN_FALLIDO", "Contraseña errónea", "0.0.0.0");
                throw new IllegalArgumentException("Contraseña incorrecta");
            }

            auditoria.registrar(username, "LOGIN_EXITOSO", "Usuario autenticado", "0.0.0.0");
            return usuario;
        } catch (Exception e) {
            auditoria.registrar(username, "ERROR_SEGURIDAD", e.getMessage(), "0.0.0.0");
            throw e;
        }
    }
}
